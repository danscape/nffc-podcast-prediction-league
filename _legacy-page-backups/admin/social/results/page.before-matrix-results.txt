import Link from "next/link";
import { createClient } from "@supabase/supabase-js";

export const dynamic = "force-dynamic";

type LatestGwResultSummaryRow = {
  fixture_id: string;
  gameweek: number;
  gameweek_label: string;
  opponent: string | null;
  opponent_short: string | null;
  venue: "H" | "A";
  home_team: string;
  away_team: string;
  home_score: number | null;
  away_score: number | null;
  forest_result: "W" | "D" | "L" | null;
  total_predictions: number | null;
  forest_win_count: number | null;
  draw_count: number | null;
  forest_loss_count: number | null;
  correct_count: number | null;
  forest_win_percent: number | string | null;
  draw_percent: number | string | null;
  forest_loss_percent: number | string | null;
  rogue_applied: boolean | null;
  maverick_applied: boolean | null;
};

type LatestGwBonusResultRow = {
  rank: number;
  fixture_id: string;
  gameweek: number;
  gameweek_label: string;
  opponent_short: string | null;
  player_id: string;
  player_display_name: string;
  player_name: string;
  short_name: string | null;
  team_id: string | null;
  team_display_name: string | null;
  team_name: string | null;
  prediction: "W" | "D" | "L" | null;
  actual_result: "W" | "D" | "L" | null;
  base_points: number | null;
  streak_bonus: number | null;
  maverick_bonus: number | null;
  rogue_bonus: number | null;
  cup_bonus: number | null;
  total_points: number | null;
  correct_streak_after_fixture: number | null;
  bonus_points: number | null;
};

type WeeklyResultRow = {
  rank: number;
  playerId: string;
  playerDisplayName: string;
  teamDisplayName: string;
  teamAbbreviation: string;
  prediction: "W" | "D" | "L" | null;
  actualResult: "W" | "D" | "L" | null;
  basePoints: number;
  streakBonus: number;
  maverickBonus: number;
  rogueBonus: number;
  cupBonus: number;
  totalPoints: number;
  bonusPoints: number;
  correctStreakAfterFixture: number;
};

function getSupabaseClient() {
  const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL;
  const supabaseAnonKey = process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY;

  if (!supabaseUrl || !supabaseAnonKey) {
    throw new Error("Missing Supabase environment variables.");
  }

  return createClient(supabaseUrl, supabaseAnonKey);
}

function formatPoints(value: number | null | undefined) {
  return Number(value ?? 0).toFixed(1).replace(".0", "");
}

function formatAverage(value: number) {
  return value.toFixed(1).replace(".0", "");
}

function formatPercent(value: number | string | null | undefined) {
  return `${Math.round(Number(value ?? 0))}%`;
}

function formatFixtureResult(summary: LatestGwResultSummaryRow | null) {
  if (!summary) return "No confirmed fixture found";

  return `${summary.gameweek_label}: ${summary.home_team} ${
    summary.home_score ?? 0
  }-${summary.away_score ?? 0} ${summary.away_team}`;
}

function getWeekModifierText(summary: LatestGwResultSummaryRow | null) {
  if (!summary) return "No scored predictions";
  if (summary.rogue_applied) return "Rogue applied";
  if (summary.maverick_applied) return "Maverick applied";
  return "No result modifier";
}

function getTeamAbbreviation(teamName: string | null | undefined) {
  const name = teamName ?? "";

  const known: Record<string, string> = {
    "1865": "1865",
    "Forest All Over": "FAO",
    "Forest Fan TV": "FFTV",
    FFTV: "FFTV",
    "Forest Focus Eternal Optimists": "FF-EO",
    "FF Eternal Optimists": "FF-EO",
    "Forest Focus Staunch Realists": "FF-SR",
    "FF Staunch Realists": "FF-SR",
    "Forest Women Cast": "FWC",
    "Oh Mist Rolling In": "OMRI",
    "Red Side Trent": "RST",
    "Talking Reds American Dream Boys": "TR-ADB",
    "TR American Dream Boys": "TR-ADB",
    "Talking Reds Brit Pop Lads": "TR-BPL",
    "TR Brit Pop Lads": "TR-BPL",
    "Tricky Talks Pod": "TT",
  };

  if (known[name]) return known[name];

  return name
    .split(/\s+/)
    .filter(Boolean)
    .map((word) => word[0])
    .join("")
    .slice(0, 6)
    .toUpperCase();
}

function getBonusChips(row: WeeklyResultRow) {
  const chips: { label: string; className: string }[] = [];

  if (row.streakBonus > 0) {
    chips.push({
      label: `S+${row.streakBonus}`,
      className: "border-amber-300 bg-amber-200/90 text-[#111111]",
    });
  }

  if (row.maverickBonus > 0) {
    chips.push({
      label: `M+${row.maverickBonus}`,
      className: "border-[#C8102E] bg-[#C8102E] text-white",
    });
  }

  if (row.rogueBonus > 0) {
    chips.push({
      label: `R+${row.rogueBonus}`,
      className: "border-white/40 bg-[#111111] text-white",
    });
  }

  if (row.cupBonus > 0) {
    chips.push({
      label: `C+${row.cupBonus}`,
      className: "border-sky-300 bg-sky-200/90 text-[#111111]",
    });
  }

  return chips;
}

function chunkRows<T>(rows: T[], parts: number) {
  const size = Math.ceil(rows.length / parts);
  return Array.from({ length: parts }, (_, index) =>
    rows.slice(index * size, index * size + size)
  );
}

export default async function AdminSocialResultsPage() {
  const supabase = getSupabaseClient();

  const [
    { data: summaryData, error: summaryError },
    { data: bonusData, error: bonusError },
  ] = await Promise.all([
    supabase.from("latest_confirmed_gw_result_summary").select("*").maybeSingle(),
    supabase
      .from("latest_confirmed_gw_bonus_results")
      .select("*")
      .order("rank", { ascending: true }),
  ]);

  if (summaryError) {
    throw new Error(summaryError.message);
  }

  if (bonusError) {
    throw new Error(bonusError.message);
  }

  const summary = (summaryData ?? null) as LatestGwResultSummaryRow | null;
  const bonusRows = (bonusData ?? []) as LatestGwBonusResultRow[];

  const weeklyRows = bonusRows
    .map((row) => {
      const teamDisplayName = row.team_display_name ?? row.team_name ?? "No team";
      const basePoints = Number(row.base_points ?? 0);
      const streakBonus = Number(row.streak_bonus ?? 0);
      const maverickBonus = Number(row.maverick_bonus ?? 0);
      const rogueBonus = Number(row.rogue_bonus ?? 0);
      const cupBonus = Number(row.cup_bonus ?? 0);
      const totalPoints = Number(row.total_points ?? 0);
      const bonusPoints = Number(row.bonus_points ?? 0);

      return {
        rank: 0,
        playerId: row.player_id,
        playerDisplayName: row.player_display_name,
        teamDisplayName,
        teamAbbreviation: getTeamAbbreviation(teamDisplayName),
        prediction: row.prediction,
        actualResult: row.actual_result,
        basePoints,
        streakBonus,
        maverickBonus,
        rogueBonus,
        cupBonus,
        totalPoints,
        bonusPoints,
        correctStreakAfterFixture: Number(row.correct_streak_after_fixture ?? 0),
      } satisfies WeeklyResultRow;
    })
    .sort((a, b) => {
      if (b.totalPoints !== a.totalPoints) return b.totalPoints - a.totalPoints;
      if (b.basePoints !== a.basePoints) return b.basePoints - a.basePoints;
      if (b.bonusPoints !== a.bonusPoints) return b.bonusPoints - a.bonusPoints;
      return a.playerDisplayName.localeCompare(b.playerDisplayName);
    })
    .map((row, index) => ({
      ...row,
      rank: index + 1,
    }));

  const totalPredictions = Number(summary?.total_predictions ?? weeklyRows.length);
  const correctPredictions = Number(
    summary?.correct_count ?? weeklyRows.filter((row) => row.basePoints > 0).length
  );
  const averageWeeklyScore =
    weeklyRows.length > 0
      ? weeklyRows.reduce((sum, row) => sum + row.totalPoints, 0) /
        weeklyRows.length
      : 0;
  const bonusPlayers = weeklyRows.filter((row) => row.bonusPoints > 0).length;
  const topScore = weeklyRows[0]?.totalPoints ?? 0;
  const topScorers = weeklyRows.filter((row) => row.totalPoints === topScore);
  const modifierText = getWeekModifierText(summary);

  return (
    <main className="min-h-screen bg-[#F7F6F2] px-4 py-6 text-[#111111] sm:px-6 lg:px-8">
      <section className="mx-auto max-w-7xl">
        <div className="mb-5 flex flex-col gap-4 md:flex-row md:items-center md:justify-between">
          <div>
            <div className="text-xs font-black uppercase tracking-[0.24em] text-[#C8102E]">
              Admin social
            </div>
            <h1 className="mt-2 text-3xl font-black uppercase tracking-tight md:text-4xl">
              Latest GW player results
            </h1>
            <p className="mt-2 text-sm font-semibold text-neutral-600">
              Auto-built from the latest confirmed fixture and weekly player scores.
            </p>
          </div>

          <Link
            href="/admin/social"
            className="rounded-full border border-[#111111] bg-white px-4 py-2 text-xs font-black uppercase tracking-wide text-[#111111] transition hover:border-[#C8102E] hover:text-[#C8102E]"
          >
            Back to social
          </Link>
        </div>

        <div className="grid gap-6">
          <section className="rounded-3xl border border-[#D9D6D1] bg-white p-4 shadow-sm md:p-5">
            <div className="mb-4">
              <h2 className="text-2xl font-black uppercase">Graphic preview</h2>
              <p className="text-sm font-semibold text-neutral-600">
                Wide layout designed for X/social sharing and full player visibility.
              </p>
            </div>

            <div className="mx-auto w-full max-w-[1120px] overflow-hidden rounded-[2rem] border border-[#111111] bg-[#111111] text-white shadow-sm">
              <div className="relative overflow-hidden bg-[radial-gradient(circle_at_top_left,_#C8102E_0,_#7A0719_35%,_#111111_78%)] p-6">
                <div className="absolute inset-0 opacity-[0.08]">
                  <div className="h-full w-full bg-[linear-gradient(135deg,_transparent_0,_transparent_47%,_#ffffff_47%,_#ffffff_53%,_transparent_53%,_transparent_100%)]" />
                </div>

                <div className="relative z-10">
                  <div className="text-[0.68rem] font-black uppercase tracking-[0.22em] text-white/75">
                    🔮 NFFC Podcast Prediction League
                  </div>

                  <h3 className="mt-2 text-5xl font-black uppercase leading-none tracking-tight">
                    {summary?.gameweek_label ?? "GW"} Results
                  </h3>

                  <div className="mt-4 grid gap-3 lg:grid-cols-[1.15fr_1fr]">
                    <div className="rounded-2xl border border-white/15 bg-white/10 p-4 backdrop-blur">
                      <div className="text-xs font-black uppercase tracking-[0.18em] text-white/60">
                        Confirmed result
                      </div>
                      <div className="mt-1 text-2xl font-black leading-tight text-white">
                        {formatFixtureResult(summary)}
                      </div>
                      <div className="mt-2 text-xs font-bold uppercase tracking-wide text-[#F4B6C1]">
                        {modifierText}
                      </div>
                    </div>

                    <div className="grid grid-cols-2 gap-2 xl:grid-cols-3">
                    <PreviewStat
                      label="Correct picks"
                      value={`${correctPredictions}/${totalPredictions}`}
                    />
                    <PreviewStat
                      label="Average score"
                      value={`${formatAverage(averageWeeklyScore)} pts`}
                    />
                    <PreviewStat
                      label="Top scorer"
                      value={
                        topScorers.length
                          ? `${topScorers[0].playerDisplayName} (${formatPoints(
                              topScore
                            )})`
                          : "TBC"
                      }
                    />
                      <PreviewStat label="Bonus players" value={bonusPlayers} />
                    </div>
                  </div>

                  <div className="mt-4 rounded-2xl border border-white/15 bg-white/10 p-4 backdrop-blur">
                    <div className="mb-2 flex items-center justify-between gap-2">
                      <div className="text-[0.68rem] font-black uppercase tracking-[0.18em] text-white/60">
                        Weekly player board
                      </div>
                      <div className="text-[0.68rem] font-black uppercase tracking-[0.18em] text-white/60">
                        Pick · Bonus · Total
                      </div>
                    </div>

                    <div className="grid grid-cols-2 gap-2">
                      {chunkRows(weeklyRows, 3).map(
                        (columnRows, columnIndex) => (
                          <div key={columnIndex} className="space-y-1.5">
                            {columnRows.map((row) => {
                              const bonusChips = getBonusChips(row);
                              const isCorrect = row.basePoints > 0;
                              const isTopScorer =
                                row.totalPoints === topScore && topScore > 0;

                              return (
                                <div
                                  key={row.playerId}
                                  className={`rounded-xl border px-2 py-2 ${
                                    isTopScorer
                                      ? "border-amber-300 bg-amber-200/15"
                                      : isCorrect
                                        ? "border-white/20 bg-white/10"
                                        : "border-white/10 bg-black/15"
                                  }`}
                                >
                                  <div className="grid grid-cols-[24px_minmax(0,1fr)_34px_86px_42px] items-center gap-1.5">
                                    <div className="text-xs font-black text-white/75">
                                      {row.rank}
                                    </div>

                                    <div className="min-w-0">
                                      <div className="truncate text-[0.74rem] font-black leading-tight text-white">
                                        {row.playerDisplayName}
                                      </div>
                                      <div className="truncate text-[0.62rem] font-bold uppercase tracking-wide text-white/55">
                                        {row.teamAbbreviation}
                                      </div>
                                    </div>

                                    <div
                                      className={`inline-flex items-center justify-center rounded-lg border px-1 py-1 text-[0.68rem] font-black ${
                                        isCorrect
                                          ? "border-green-300 bg-green-400 text-[#111111]"
                                          : "border-white/20 bg-white/10 text-white"
                                      }`}
                                    >
                                      {row.prediction ?? "—"}
                                    </div>

                                    <div className="flex min-h-[24px] flex-wrap items-center gap-1">
                                      {bonusChips.length ? (
                                        bonusChips.map((chip) => (
                                          <span
                                            key={chip.label}
                                            className={`inline-flex items-center justify-center rounded-md border px-1.5 py-0.5 text-[0.56rem] font-black ${chip.className}`}
                                          >
                                            {chip.label}
                                          </span>
                                        ))
                                      ) : (
                                        <span className="text-[0.62rem] font-bold text-white/35">
                                          —
                                        </span>
                                      )}
                                    </div>

                                    <div className="text-right">
                                      <span className="inline-flex min-w-[34px] items-center justify-center rounded-lg bg-white px-1.5 py-1 text-[0.68rem] font-black text-[#111111]">
                                        {formatPoints(row.totalPoints)}
                                      </span>
                                    </div>
                                  </div>
                                </div>
                              );
                            })}
                          </div>
                        )
                      )}
                    </div>
                  </div>

                  <div className="mt-4 rounded-2xl border border-white/15 bg-white/10 p-3 text-[0.62rem] font-bold uppercase tracking-wide text-white/70 backdrop-blur">
                    S = streak bonus · M = maverick bonus · R = rogue bonus · C =
                    cup bonus
                  </div>
                </div>
              </div>
            </div>
          </section>

          <section className="rounded-3xl border border-[#D9D6D1] bg-white p-4 shadow-sm md:p-5">
            <div className="mb-4">
              <h2 className="text-2xl font-black uppercase">Weekly results data</h2>
              <p className="text-sm font-semibold text-neutral-600">
                Plain admin view of the same result set used in the graphic.
              </p>
            </div>

            <div className="mb-4 grid gap-3 md:grid-cols-4">
              <AdminStatCard
                label="Fixture"
                value={summary?.gameweek_label ?? "TBC"}
              />
              <AdminStatCard
                label="Result"
                value={`${summary?.home_score ?? 0}-${summary?.away_score ?? 0}`}
              />
              <AdminStatCard
                label="Correct picks"
                value={`${correctPredictions}/${totalPredictions}`}
              />
              <AdminStatCard label="Bonus players" value={bonusPlayers} />
            </div>

            <div className="overflow-hidden rounded-2xl border border-[#D9D6D1]">
              <table className="w-full border-collapse text-left text-xs">
                <thead>
                  <tr>
                    <th className="bg-[#111111] px-3 py-3 text-white">POS</th>
                    <th className="bg-[#111111] px-3 py-3 text-white">
                      PLAYER
                    </th>
                    <th className="bg-[#111111] px-3 py-3 text-white">TEAM</th>
                    <th className="bg-[#111111] px-3 py-3 text-center text-white">
                      PICK
                    </th>
                    <th className="bg-[#111111] px-3 py-3 text-center text-white">
                      BASE
                    </th>
                    <th className="bg-[#111111] px-3 py-3 text-white">BONUS</th>
                    <th className="bg-[#111111] px-3 py-3 text-center text-white">
                      TOTAL
                    </th>
                  </tr>
                </thead>

                <tbody>
                  {weeklyRows.map((row) => {
                    const bonusChips = getBonusChips(row);

                    return (
                      <tr
                        key={row.playerId}
                        className={`border-b border-[#E7E2DA] last:border-b-0 ${
                          row.basePoints > 0 ? "bg-[#F7FAF7]" : "bg-white"
                        }`}
                      >
                        <td className="px-3 py-3 text-sm font-black text-[#111111]">
                          {row.rank}
                        </td>
                        <td className="px-3 py-3">
                          <div className="font-black text-[#111111]">
                            {row.playerDisplayName}
                          </div>
                        </td>
                        <td className="px-3 py-3">
                          <div className="font-bold text-neutral-700">
                            {row.teamDisplayName}
                          </div>
                        </td>
                        <td className="px-3 py-3 text-center">
                          <span
                            className={`inline-flex min-w-[30px] items-center justify-center rounded-lg px-2 py-1 text-xs font-black ${
                              row.basePoints > 0
                                ? "bg-green-500 text-[#111111]"
                                : "bg-[#F1F1F1] text-[#111111]"
                            }`}
                          >
                            {row.prediction ?? "—"}
                          </span>
                        </td>
                        <td className="px-3 py-3 text-center font-black">
                          {row.basePoints}
                        </td>
                        <td className="px-3 py-3">
                          <div className="flex flex-wrap gap-1">
                            {bonusChips.length ? (
                              bonusChips.map((chip) => (
                                <span
                                  key={chip.label}
                                  className={`inline-flex items-center justify-center rounded-md border px-2 py-1 text-[0.68rem] font-black ${chip.className}`}
                                >
                                  {chip.label}
                                </span>
                              ))
                            ) : (
                              <span className="text-neutral-400">—</span>
                            )}
                          </div>
                        </td>
                        <td className="px-3 py-3 text-center">
                          <span className="inline-flex min-w-[40px] items-center justify-center rounded-lg bg-[#111111] px-2 py-1 text-xs font-black text-white">
                            {formatPoints(row.totalPoints)}
                          </span>
                        </td>
                      </tr>
                    );
                  })}
                </tbody>
              </table>
            </div>

            <div className="mt-4 rounded-2xl border border-[#D9D6D1] bg-[#F7F6F2] p-4">
              <div className="text-sm font-black uppercase text-[#111111]">
                Copy-ready X post starter
              </div>
              <div className="mt-2 whitespace-pre-wrap text-sm font-semibold leading-6 text-neutral-700">
                {`🔮 ${summary?.gameweek_label ?? "GW"} results are in.

${formatFixtureResult(summary)}

Top scorer${topScorers.length === 1 ? "" : "s"}: ${
                  topScorers.length
                    ? topScorers
                        .map(
                          (row) =>
                            `${row.playerDisplayName} (${formatPoints(
                              row.totalPoints
                            )})`
                        )
                        .join(", ")
                    : "TBC"
                }
Correct picks: ${correctPredictions}/${totalPredictions}
Bonus players: ${bonusPlayers}
${modifierText}

Full board below 👇`}
              </div>
            </div>
          </section>
        </div>
      </section>
    </main>
  );
}

function PreviewStat({
  label,
  value,
}: {
  label: string;
  value: string | number;
}) {
  return (
    <div className="rounded-2xl border border-white/15 bg-white/10 p-3 backdrop-blur">
      <div className="text-[0.62rem] font-black uppercase tracking-[0.18em] text-white/60">
        {label}
      </div>
      <div className="mt-1 text-sm font-black leading-tight text-white">
        {value}
      </div>
    </div>
  );
}

function AdminStatCard({
  label,
  value,
}: {
  label: string;
  value: string | number;
}) {
  return (
    <div className="rounded-2xl border border-[#D9D6D1] bg-[#F7F6F2] p-4">
      <div className="text-xs font-black uppercase tracking-wide text-[#C8102E]">
        {label}
      </div>
      <div className="mt-1 text-lg font-black text-[#111111]">{value}</div>
    </div>
  );
}
