function getSupabaseClient() {
  const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL;
  const supabaseAnonKey = process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY;

  if (!supabaseUrl || !supabaseAnonKey) {
    throw new Error("Missing Supabase environment variables.");
  }

  return createClient(supabaseUrl, supabaseAnonKey);
}

function formatPoints(value: number | null | undefined) {
  return Number(value ?? 0).toFixed(1).replace(".0", "");
}

function formatAverage(value: number) {
  return value.toFixed(1).replace(".0", "");
}

function formatPercent(value: number | string | null | undefined) {
  return `${Math.round(Number(value ?? 0))}%`;
}

function formatFixtureResult(summary: LatestGwResultSummaryRow | null) {
  if (!summary) return "No confirmed fixture found";

  return `${summary.gameweek_label}: ${summary.home_team} ${
    summary.home_score ?? 0
  }-${summary.away_score ?? 0} ${summary.away_team}`;
}

function resultLabel(result: "W" | "D" | "L" | null) {
  if (result === "W") return "Forest win";
  if (result === "D") return "Draw";
  if (result === "L") return "Forest defeat";
  return "Result TBC";
}

function getWeekModifierText(summary: LatestGwResultSummaryRow | null) {
  if (!summary) return "No scored predictions";
  if (summary.rogue_applied) return "Rogue applied";
  if (summary.maverick_applied) return "Maverick applied";
  return "No result modifier";
}

function getTeamAbbreviation(teamName: string | null | undefined) {
  const name = teamName ?? "";

  const known: Record<string, string> = {
    "1865": "1865",
    "Forest All Over": "FAO",
    "Forest Fan TV": "FFTV",
    FFTV: "FFTV",
    "Forest Focus Eternal Optimists": "FF-EO",
    "FF Eternal Optimists": "FF-EO",
    "Forest Focus Staunch Realists": "FF-SR",
    "FF Staunch Realists": "FF-SR",
    "Forest Women Cast": "FWC",
    "Oh Mist Rolling In": "OMRI",
    "Red Side Trent": "RST",
    "Talking Reds American Dream Boys": "TR-ADB",
    "TR American Dream Boys": "TR-ADB",
    "Talking Reds Brit Pop Lads": "TR-BPL",
    "TR Brit Pop Lads": "TR-BPL",
    "Tricky Talks Pod": "TT",
  };

  if (known[name]) return known[name];

  return name
    .split(/\s+/)
    .filter(Boolean)
    .map((word) => word[0])
    .join("")
    .slice(0, 6)
    .toUpperCase();
}

function getTeamOrder(teamName: string) {
  const index = TEAM_ORDER.indexOf(teamName);
  return index === -1 ? 999 : index;
}

function getMostCommonPick(rows: WeeklyResultRow[]) {
  const counts = {
    W: rows.filter((row) => row.prediction === "W").length,
    D: rows.filter((row) => row.prediction === "D").length,
    L: rows.filter((row) => row.prediction === "L").length,
  };

  const sorted = Object.entries(counts).sort((a, b) => b[1] - a[1]);
  const [pick, count] = sorted[0] ?? ["W", 0];

  const label =
    pick === "W" ? "Forest win" : pick === "D" ? "Draw" : "Forest defeat";

  const percentage = rows.length ? Math.round((count / rows.length) * 100) : 0;

  return {
    label,
    count,
    percentage,
  };
}

function getBonusChips(row: WeeklyResultRow) {
  const chips: { label: string; className: string }[] = [];

  if (row.streakBonus > 0) {
    chips.push({
      label: `S+${row.streakBonus}`,
      className: "border-yellow-200 bg-yellow-300 text-[#111111]",
    });
  }

  if (row.maverickBonus > 0) {
    chips.push({
      label: `M+${row.maverickBonus}`,
      className: "border-sky-200 bg-sky-400 text-[#06121F]",
    });
  }

  if (row.rogueBonus > 0) {
    chips.push({
      label: `R+${row.rogueBonus}`,
      className: "border-pink-200 bg-pink-500 text-white",
    });
  }

  if (row.cupBonus > 0) {
    chips.push({
      label: `C+${row.cupBonus}`,
      className: "border-purple-200 bg-purple-400 text-white",
    });
  }

  return chips;
}

function buildTeamGroups(rows: WeeklyResultRow[]) {
  const map = new Map<string, WeeklyResultRow[]>();

  rows.forEach((row) => {
    const existing = map.get(row.teamDisplayName) ?? [];
    existing.push(row);
    map.set(row.teamDisplayName, existing);
  });

  return Array.from(map.entries())
    .map(([teamName, teamRows]) => {
      const sortedRows = [...teamRows].sort((a, b) =>
        a.playerDisplayName.localeCompare(b.playerDisplayName)
      );

      return {
        teamName,
        abbreviation: getTeamAbbreviation(teamName),
        rows: sortedRows,
        playerCount: sortedRows.length,
        correctCount: sortedRows.filter((row) => row.basePoints > 0).length,
        bonusPlayers: sortedRows.filter((row) => row.bonusPoints > 0).length,
        totalPlayerPoints: sortedRows.reduce(
          (total, row) => total + row.totalPoints,
          0
        ),
      } satisfies TeamGroup;
    })
    .sort((a, b) => {
      const orderDifference = getTeamOrder(a.teamName) - getTeamOrder(b.teamName);
      if (orderDifference !== 0) return orderDifference;
      return a.teamName.localeCompare(b.teamName);
    });
}

}) {
  return (

      <div className="text-[0.68rem] font-black uppercase tracking-[0.18em] text-white/55">

      </div>
      <div className="mt-1 text-2xl font-black leading-tight text-white">

}

    <div className="min-h-[104px] border-r border-t border-white/10 p-2 text-center last:border-r-0">
      <div className="truncate text-[0.82rem] font-black leading-tight text-white">
        {row.playerDisplayName}
      </div>

      <div className="mt-2 flex items-center justify-center gap-1.5">
        <span
          className={`flex h-7 w-7 items-center justify-center rounded-full text-base font-black ${
            isCorrect ? "bg-green-500 text-white" : "bg-white/20 text-white/70"
          }`}
        >
          {isCorrect ? "✓" : "×"}
        </span>

        {row.basePoints > 0 && (
          <span className="rounded-lg bg-green-500 px-2 py-1 text-[0.62rem] font-black text-white">
            +{row.basePoints}
          </span>
        )}
      </div>

      <div className="mt-2 flex min-h-[22px] flex-wrap items-center justify-center gap-1">
        {bonusChips.length ? (
          bonusChips.map((chip) => (
            <span
              key={chip.label}
              className={`inline-flex items-center justify-center rounded-md border px-1.5 py-0.5 text-[0.56rem] font-black ${chip.className}`}
            >
              {chip.label}
            </span>
          ))
        ) : (
          <span className="text-[0.6rem] font-bold text-white/30">—</span>
        )}
      </div>

      <div className="mt-2">
        <span
          className={`inline-flex min-w-[32px] items-center justify-center rounded-lg px-2 py-0.5 text-[0.62rem] font-black ${
            row.totalPoints > 0
              ? "bg-green-500 text-white"
              : "bg-white/10 text-white/45"
          }`}
        >
          {formatPoints(row.totalPoints)}
        </span>
      </div>
    </div>
  );
}
