type TeamGroup = {
  key: string;
  teamName: string;
  abbreviation: string | null;
  xHandle: string | null;
  teamResult: LatestTeamResultRow | null;
  players: LatestBonusResultRow[];
};

const resultLabels: Record<PredictionValue, string> = {
  W: "Forest win",
  D: "Draw",
  L: "Forest defeat",
};

function toNumber(value: number | string | null | undefined) {
  const numericValue = Number(value ?? 0);
  return Number.isFinite(numericValue) ? numericValue : 0;
}

function formatPoints(value: number | string | null | undefined) {
  return toNumber(value).toFixed(1).replace(".0", "");
}

function formatPercent(value: number | string | null | undefined) {
  return `${Math.round(toNumber(value))}%`;
}

function formatDateTime(value: string | null | undefined) {
  if (!value) return "DATE TBC";

  return new Intl.DateTimeFormat("en-GB", {
    weekday: "short",
    day: "numeric",
    month: "short",
    hour: "2-digit",
    minute: "2-digit",
  })
    .format(new Date(value))
    .toUpperCase();
}

function getFixtureId(summary: LatestResultSummaryRow | null) {
  return summary?.fixture_id ?? summary?.id ?? "latest-confirmed-fixture";
}

function getActualResult(summary: LatestResultSummaryRow | null) {
  return summary?.forest_result ?? summary?.actual_result ?? null;
}

function getOpponentName(summary: LatestResultSummaryRow | null) {
  return summary?.opponent_short ?? summary?.opponent ?? "OPP";
}

function getGameweekLabel(summary: LatestResultSummaryRow | null) {
  return summary?.gameweek_label ?? (summary?.gameweek ? `GW${summary.gameweek}` : "LATEST GW");
}

function getTotalPredictions(summary: LatestResultSummaryRow | null) {
  return (
    summary?.total_predictions ??
    summary?.total_players ??
    summary?.prediction_count ??
    0
  );
}

function getCorrectCount(summary: LatestResultSummaryRow | null) {
  return summary?.correct_count ?? 0;
}

function getCorrectPercentage(summary: LatestResultSummaryRow | null) {
  const explicit = summary?.correct_percentage ?? summary?.correct_pick_percentage;

  if (explicit !== null && explicit !== undefined) {
    return toNumber(explicit);
  }

  const total = getTotalPredictions(summary);
  const correct = getCorrectCount(summary);

  return total ? (correct / total) * 100 : 0;
}

function getPredictionSplit(summary: LatestResultSummaryRow | null) {
  const forestWinCount = summary?.forest_win_count ?? summary?.win_count ?? 0;
  const drawCount = summary?.draw_count ?? 0;
  const forestLossCount = summary?.forest_loss_count ?? summary?.loss_count ?? 0;
  const total = getTotalPredictions(summary) || forestWinCount + drawCount + forestLossCount;

  return {
    forestWinCount,
    drawCount,
    forestLossCount,
    forestWinPercent: total ? (forestWinCount / total) * 100 : 0,
    drawPercent: total ? (drawCount / total) * 100 : 0,
    forestLossPercent: total ? (forestLossCount / total) * 100 : 0,
  };
}

function getScoreText(summary: LatestResultSummaryRow | null) {
  if (!summary) return "NO CONFIRMED RESULT";

  if (summary.home_score === null || summary.home_score === undefined) {
    return `${getGameweekLabel(summary)} // FOREST ${summary.venue === "A" ? "AT" : "V"} ${getOpponentName(summary)}`;
  }

  return `${getGameweekLabel(summary)} // ${summary.home_team ?? "HOME"} ${summary.home_score}-${summary.away_score ?? "?"} ${summary.away_team ?? "AWAY"}`;
}

function getResultLabel(summary: LatestResultSummaryRow | null) {
  if (!summary) return "RESULT TBC";

  if (summary.result_label) return summary.result_label;

  const actualResult = getActualResult(summary);
  return actualResult ? resultLabels[actualResult].toUpperCase() : "RESULT TBC";
}

function getPlayerName(row: LatestBonusResultRow) {
  return row.table_display_name ?? row.short_name ?? row.player_name ?? "Unknown";
}

function getTeamName(row: LatestBonusResultRow | LatestTeamResultRow) {
  return row.team_display_name ?? row.display_name ?? row.team_name ?? "Unassigned";
}

function getTeamAbbreviation(row: LatestBonusResultRow | LatestTeamResultRow) {
  return row.team_abbreviation ?? row.abbreviation ?? null;
}

function getTeamKey(row: LatestBonusResultRow | LatestTeamResultRow) {
  return row.team_id ?? getTeamName(row).toLowerCase().replace(/[^a-z0-9]+/g, "-");
}

function getPlayerTotal(row: LatestBonusResultRow) {
  return toNumber(row.total_points);
}

function getBasePoints(row: LatestBonusResultRow) {
  return toNumber(row.base_points);
}

function getBonusTotal(row: LatestBonusResultRow) {
  const explicit = row.bonus_points;

  if (explicit !== null && explicit !== undefined) {
    return toNumber(explicit);
  }

  return (
    toNumber(row.streak_bonus) +
    toNumber(row.maverick_bonus) +
    toNumber(row.rogue_bonus) +
    toNumber(row.cup_bonus)
  );
}

function isCorrectPick(row: LatestBonusResultRow, summary: LatestResultSummaryRow | null) {
  if (row.is_correct !== null && row.is_correct !== undefined) return Boolean(row.is_correct);
  if (row.correct !== null && row.correct !== undefined) return Boolean(row.correct);

  const actualResult = row.actual_result ?? row.forest_result ?? getActualResult(summary);

  if (row.prediction && actualResult) return row.prediction === actualResult;

  return getBasePoints(row) > 0;
}

function getTeamWeeklyPoints(row: LatestTeamResultRow | null | undefined) {
  if (!row) return 0;

  return toNumber(row.team_points ?? row.points_this_week ?? row.weekly_team_points);
}

function getTeamCorrectCount(row: LatestTeamResultRow | null | undefined) {
  return toNumber(row?.players_correct ?? row?.correct_count);
}

function getTeamPlayerCount(row: LatestTeamResultRow | null | undefined) {
  return toNumber(row?.player_count ?? row?.total_players);
}

function getTeamCorrectPercentage(row: LatestTeamResultRow | null | undefined) {
  const explicit = row?.correct_percentage ?? row?.accuracy_percentage;

  if (explicit !== null && explicit !== undefined) return toNumber(explicit);

  const playerCount = getTeamPlayerCount(row);

  return playerCount ? (getTeamCorrectCount(row) / playerCount) * 100 : 0;
}

function getTopTeam(teamRows: LatestTeamResultRow[]) {
  return [...teamRows].sort((a, b) => {
    const pointsDifference = getTeamWeeklyPoints(b) - getTeamWeeklyPoints(a);
    if (pointsDifference !== 0) return pointsDifference;

    const percentDifference = getTeamCorrectPercentage(b) - getTeamCorrectPercentage(a);
    if (percentDifference !== 0) return percentDifference;

    return getTeamName(a).localeCompare(getTeamName(b));
  })[0] ?? null;
}

function getTopScorers(playerRows: LatestBonusResultRow[]) {
  const bestScore = Math.max(0, ...playerRows.map(getPlayerTotal));

  return playerRows.filter((row) => getPlayerTotal(row) === bestScore && bestScore > 0);
}

function buildTeamGroups(
  playerRows: LatestBonusResultRow[],
  teamRows: LatestTeamResultRow[]
): TeamGroup[] {
  const teamResultMap = new Map<string, LatestTeamResultRow>();

  teamRows.forEach((team) => {
    teamResultMap.set(getTeamKey(team), team);
  });

  const groupMap = new Map<string, TeamGroup>();

  playerRows.forEach((player) => {
    const key = getTeamKey(player);
    const existing = groupMap.get(key);

    if (existing) {
      existing.players.push(player);
      return;
    }

    const matchingTeamResult = teamResultMap.get(key) ?? null;

    groupMap.set(key, {
      key,
      teamName: getTeamName(player),
      abbreviation: getTeamAbbreviation(player),
      xHandle: matchingTeamResult?.x_handle ?? null,
      teamResult: matchingTeamResult,
      players: [player],
    });
  });

  teamRows.forEach((team) => {
    const key = getTeamKey(team);
    if (groupMap.has(key)) return;

    groupMap.set(key, {
      key,
      teamName: getTeamName(team),
      abbreviation: getTeamAbbreviation(team),
      xHandle: team.x_handle ?? null,
      teamResult: team,
      players: [],
    });
  });

  return Array.from(groupMap.values()).sort((a, b) => {
    const pointsDifference = getTeamWeeklyPoints(b.teamResult) - getTeamWeeklyPoints(a.teamResult);
    if (pointsDifference !== 0) return pointsDifference;

    const percentDifference =
      getTeamCorrectPercentage(b.teamResult) - getTeamCorrectPercentage(a.teamResult);
    if (percentDifference !== 0) return percentDifference;

    const sortA = a.teamResult?.sort_order ?? 999;
    const sortB = b.teamResult?.sort_order ?? 999;
    if (sortA !== sortB) return sortA - sortB;

    return a.teamName.localeCompare(b.teamName);
  });
}

function buildWeeklyResultPost({
  summary,
  playerRows,
  teamRows,
}: {
  summary: LatestResultSummaryRow | null;
  playerRows: LatestBonusResultRow[];
  teamRows: LatestTeamResultRow[];
}) {
  if (!summary) {
    return [
      "🔮 NFFC Podcast Prediction League",
      "",
      "Latest weekly results will appear once a confirmed fixture is available.",
      "",
      "#NFFC",
    ].join("\n");
  }

  const topScorers = getTopScorers(playerRows);
  const topTeam = getTopTeam(teamRows);
  const topScorerText = topScorers.length
    ? topScorers.map(getPlayerName).join(" + ")
    : "TBC";

  const bonusLine = summary.rogue_applied
    ? "💗 Rogue bonus was active this GW."
    : summary.maverick_applied
      ? "🩵 Maverick bonus was active this GW."
      : "No maverick or rogue bonus this GW.";

  return [
    "🔮 NFFC Podcast Prediction League",
    "",
    getScoreText(summary),
    `Result: ${getResultLabel(summary)}`,
    "",
    `🏆 Top scorer${topScorers.length === 1 ? "" : "s"}: ${topScorerText} — ${formatPoints(topScorers[0]?.total_points)} pts`,
    topTeam
      ? `🏆 Team of the week: ${getTeamName(topTeam)} — ${formatPercent(getTeamCorrectPercentage(topTeam))} correct / ${formatPoints(getTeamWeeklyPoints(topTeam))} pts`
      : "🏆 Team of the week: TBC",
    bonusLine,
    "",
    "#NFFC",
  ].join("\n");
}

    void loadResultsData();

function ResultsMatrix({

  summary,

  topScorers,

}: {
  fixtureId: string;
  summary: LatestResultSummaryRow | null;
  teamGroups: TeamGroup[];
  topScorers: LatestBonusResultRow[];
  topTeam: LatestTeamResultRow | null;
  predictionSplit: ReturnType<typeof getPredictionSplit>;
}) {
  const hasRows = teamGroups.some((group) => group.players.length > 0);

  return (
    <div
      id={`results-matrix-${fixtureId}`}
      className="overflow-hidden border border-[var(--nffc-white,#f5f5f5)] bg-[var(--nffc-black,#000000)] text-[var(--nffc-white,#f5f5f5)]"
    >
      <div className="grid gap-px bg-[var(--nffc-white,#f5f5f5)]">
        <div className="bg-[var(--nffc-red,#e50914)] px-3 py-2 text-xs font-black uppercase tracking-[0.2em] text-white md:text-sm">
          NFFC Podcast Prediction League // Weekly results terminal
        </div>

        <div className="grid gap-px bg-[var(--nffc-white,#f5f5f5)] md:grid-cols-[1fr_auto]">
          <div className="bg-[var(--nffc-black,#000000)] p-3 md:p-4">
            <div className="text-[0.72rem] font-black uppercase tracking-[0.24em] text-[var(--stat-yellow,#ffe44d)]">
              {`${getGameweekLabel(summary)} // Latest confirmed result`}
            </div>
            <h1 className="mt-2 text-2xl font-black uppercase leading-tight text-[var(--nffc-white,#f5f5f5)] md:text-4xl">
              {getScoreText(summary)}
            </h1>
          </div>

          <div className="grid min-w-[260px] content-center bg-[var(--nffc-black,#000000)] p-3 md:p-4">
            <ResultBadge>{getResultLabel(summary)}</ResultBadge>
          </div>
        </div>

        <div className="grid gap-px bg-[var(--nffc-white,#f5f5f5)] lg:grid-cols-4">
          <BroadcastStat label="Correct picks" value={`${getCorrectCount(summary)}/${getTotalPredictions(summary)}`} tone="green" />
          <BroadcastStat label="Correct share" value={formatPercent(getCorrectPercentage(summary))} tone="yellow" />
          <BroadcastStat
            label="Top scorer"
            value={
              topScorers.length
                ? `${topScorers.map(getPlayerName).join(" + ")} / ${formatPoints(topScorers[0]?.total_points)}`
                : "TBC"
            }
            tone="white"
          />
          <BroadcastStat
            label="Team of week"
            value={topTeam ? `${getTeamName(topTeam)} / ${formatPoints(getTeamWeeklyPoints(topTeam))}` : "TBC"}
            tone="cyan"
          />
        </div>

        <div className="grid gap-px bg-[var(--nffc-white,#f5f5f5)] lg:grid-cols-[1fr_1fr_1fr_auto]">
          <PredictionSplitCell label="Forest win" count={predictionSplit.forestWinCount} percent={predictionSplit.forestWinPercent} tone="green" />
          <PredictionSplitCell label="Draw" count={predictionSplit.drawCount} percent={predictionSplit.drawPercent} tone="yellow" />
          <PredictionSplitCell label="Forest defeat" count={predictionSplit.forestLossCount} percent={predictionSplit.forestLossPercent} tone="red" />
          <BonusStatusCell summary={summary} />
        </div>
      </div>

      {!hasRows ? (
        <div className="border-t border-[var(--nffc-white,#f5f5f5)] bg-[var(--nffc-black,#000000)] p-8 text-center text-sm font-black uppercase tracking-[0.18em] text-[var(--nffc-muted,#a7a7a7)]">
          No latest confirmed player result rows found.
        </div>
      ) : (
        <div className="grid gap-px bg-[var(--nffc-white,#f5f5f5)] border-t border-[var(--nffc-white,#f5f5f5)]">
          {teamGroups.map((group, index) => (
            <TeamResultBlock key={group.key} group={group} rank={index + 1} summary={summary} />
          ))}
        </div>
      )}
    </div>
  );
}

function TeamResultBlock({
  group,
  rank,
  summary,
}: {
  group: TeamGroup;
  rank: number;
  summary: LatestResultSummaryRow | null;
}) {
  return (
    <section className="bg-[var(--nffc-panel,#070707)]">
      <div className="grid gap-px bg-[var(--nffc-white,#f5f5f5)] md:grid-cols-[72px_1fr_auto]">
        <div className="flex items-center justify-center bg-[var(--nffc-red-dark,#8f0612)] px-3 py-2 text-xl font-black text-white">
          {String(rank).padStart(2, "0")}
        </div>
        <div className="bg-[var(--nffc-black,#000000)] px-3 py-2">
          <div className="flex flex-wrap items-center gap-2">
            <h2 className="text-lg font-black uppercase tracking-tight text-[var(--nffc-white,#f5f5f5)]">
              {group.teamName}
            </h2>
            {group.abbreviation && (
              <TeletextPill tone="muted">{group.abbreviation}</TeletextPill>
            )}
            {group.xHandle && <TeletextPill tone="cyan">{group.xHandle}</TeletextPill>}
          </div>
        </div>
        <div className="bg-[var(--nffc-black,#000000)] px-3 py-2 text-right text-[0.68rem] font-black uppercase tracking-[0.18em] text-[var(--nffc-muted,#a7a7a7)]">
          Team row // weekly scoring view
        </div>
      </div>

      <div className="grid gap-px bg-[var(--nffc-white,#f5f5f5)] p-px sm:grid-cols-2 lg:grid-cols-3 2xl:grid-cols-4">
        {group.players.length ? (
          group.players
            .sort((a, b) => {
              const pointsDifference = getPlayerTotal(b) - getPlayerTotal(a);
              if (pointsDifference !== 0) return pointsDifference;

              const correctDifference = Number(isCorrectPick(b, summary)) - Number(isCorrectPick(a, summary));
              if (correctDifference !== 0) return correctDifference;

              return getPlayerName(a).localeCompare(getPlayerName(b));
            })
            .map((player) => (
              <PlayerResultCell
                key={`${group.key}-${player.player_id ?? getPlayerName(player)}`}
                player={player}
                summary={summary}
              />
            ))
        ) : (
          <div className="bg-[var(--nffc-black,#000000)] p-4 text-sm font-bold uppercase tracking-[0.14em] text-[var(--nffc-muted,#a7a7a7)] sm:col-span-2 lg:col-span-3 2xl:col-span-4">
            No player rows returned for this team.
          </div>
        )}
      </div>
    </section>
  );
}

function PlayerResultCell({
  player,
  summary,
}: {
  player: LatestBonusResultRow;
  summary: LatestResultSummaryRow | null;
}) {
  const correct = isCorrectPick(player, summary);
  const basePoints = getBasePoints(player);
  const streakBonus = toNumber(player.streak_bonus);
  const maverickBonus = toNumber(player.maverick_bonus);
  const rogueBonus = toNumber(player.rogue_bonus);
  const cupBonus = toNumber(player.cup_bonus);
  const totalPoints = getPlayerTotal(player);

  return (
    <article className="min-h-[128px] bg-[var(--nffc-black,#000000)] p-3">
      <div className="flex items-start justify-between gap-3">
        <div className="min-w-0">
          <div className="truncate text-base font-black uppercase text-[var(--nffc-white,#f5f5f5)]">
            {getPlayerName(player)}
          </div>
          <div className="mt-1 flex flex-wrap items-center gap-1 text-[0.68rem] font-black uppercase tracking-[0.14em] text-[var(--nffc-muted,#a7a7a7)]">
            <span>Pick</span>
            <span className="border border-[var(--nffc-muted,#a7a7a7)] px-1.5 py-0.5 text-[var(--nffc-white,#f5f5f5)]">
              {player.prediction ?? "-"}
            </span>
          </div>
        </div>

        <OutcomeMark correct={correct} />
      </div>

      <div className="mt-3 flex flex-wrap gap-1.5">
        {basePoints > 0 && <ScoreChip tone="green" label={`B+${formatPoints(basePoints)}`} />}
        {streakBonus > 0 && <ScoreChip tone="yellow" label={`S+${formatPoints(streakBonus)}`} />}
        {maverickBonus > 0 && <ScoreChip tone="cyan" label={`M+${formatPoints(maverickBonus)}`} />}
        {rogueBonus > 0 && <ScoreChip tone="pink" label={`R+${formatPoints(rogueBonus)}`} />}
        {cupBonus > 0 && <ScoreChip tone="green" label={`C+${formatPoints(cupBonus)}`} />}
        {getBonusTotal(player) <= 0 && basePoints <= 0 && (
          <ScoreChip tone="red" label="MISS" />
        )}
      </div>

      <div className="mt-3 flex justify-end">
        <ScoreChip tone="white" label={`T ${formatPoints(totalPoints)}`} large />
      </div>
    </article>
  );
}

function OutcomeMark({ correct }: { correct: boolean }) {
  return (
    <div
      className={`flex h-9 w-9 shrink-0 items-center justify-center border bg-[var(--nffc-black,#000000)] text-xl font-black ${
        correct
          ? "border-[var(--stat-green,#22e55e)] text-[var(--stat-green,#22e55e)]"
          : "border-[var(--stat-wrong,#ff3030)] text-[var(--stat-wrong,#ff3030)]"
      }`}
      aria-label={correct ? "Correct pick" : "Wrong pick"}
    >
      {correct ? "✓" : "×"}
    </div>
  );
}

function ScoreChip({
  label,
  tone,
  large = false,
}: {
  label: string;
  tone: Tone | "red";
  large?: boolean;
}) {
  const toneClass = getToneClass(tone);

  return (
    <span
      className={`inline-flex items-center border bg-[var(--nffc-black,#000000)] font-black uppercase tracking-[0.12em] ${toneClass} ${
        large ? "px-3 py-1.5 text-sm" : "px-2 py-1 text-[0.68rem]"
      }`}
    >
      {label}
    </span>
  );
}

function ResultBadge({ children }: { children: ReactNode }) {
  return (
    <div className="w-full border border-[var(--nffc-white,#f5f5f5)] bg-[var(--nffc-black,#000000)] px-4 py-3 text-center text-sm font-black uppercase tracking-[0.18em] text-[var(--nffc-white,#f5f5f5)]">
      {children}
    </div>
  );
}

function BroadcastStat({ label, value, tone }: { label: string; value: string; tone: Tone }) {
  return (
    <div className="bg-[var(--nffc-black,#000000)] p-3">
      <div className="text-[0.68rem] font-black uppercase tracking-[0.18em] text-[var(--nffc-muted,#a7a7a7)]">
        {label}
      </div>
      <div className={`mt-1 truncate text-lg font-black uppercase ${getTextToneClass(tone)}`}>
        {value}
      </div>
    </div>
  );
}

function PredictionSplitCell({
  label,
  count,
  percent,
  tone,
}: {
  label: string;
  count: number;
  percent: number;
  tone: Tone | "red";
}) {
  return (
    <div className="bg-[var(--nffc-black,#000000)] p-3">
      <div className="flex items-center justify-between gap-3">
        <div className="text-[0.72rem] font-black uppercase tracking-[0.18em] text-[var(--nffc-muted,#a7a7a7)]">
          {label}
        </div>
        <ScoreChip tone={tone} label={`${count}`} />
      </div>
      <div className={`mt-2 text-3xl font-black leading-none ${getTextToneClass(tone)}`}>
        {formatPercent(percent)}
      </div>
    </div>
  );
}

function BonusStatusCell({ summary }: { summary: LatestResultSummaryRow | null }) {
  const rogueApplied = Boolean(summary?.rogue_applied);
  const maverickApplied = Boolean(summary?.maverick_applied) && !rogueApplied;

  return (
    <div className="grid min-w-[220px] content-center gap-2 bg-[var(--nffc-black,#000000)] p-3">
      <ScoreChip
        tone={maverickApplied ? "cyan" : "muted"}
        label={maverickApplied ? "MAVERICK APPLIED" : "NO MAVERICK"}
      />
      <ScoreChip
        tone={rogueApplied ? "pink" : "muted"}
        label={rogueApplied ? "ROGUE APPLIED" : "NO ROGUE"}
      />
    </div>
  );
}

function InfoRow({ label, value }: { label: string; value: ReactNode }) {
  return (
    <div className="grid grid-cols-[110px_1fr] gap-px border-b border-[var(--nffc-white,#f5f5f5)] last:border-b-0">
      <div className="bg-[var(--nffc-red-dark,#8f0612)] px-2 py-2 text-[0.68rem] font-black uppercase tracking-[0.14em] text-white">
        {label}
      </div>
      <div className="bg-[var(--nffc-black,#000000)] px-2 py-2 text-[var(--nffc-white,#f5f5f5)]">
        {value}
      </div>
    </div>
  );
}

function getToneClass(tone: Tone | "red") {
  if (tone === "green") return "border-[var(--stat-green,#22e55e)] text-[var(--stat-green,#22e55e)]";
  if (tone === "yellow") return "border-[var(--stat-yellow,#ffe44d)] text-[var(--stat-yellow,#ffe44d)]";
  if (tone === "cyan") return "border-[var(--stat-cyan,#59efff)] text-[var(--stat-cyan,#59efff)]";
  if (tone === "pink") return "border-[var(--stat-pink,#ff4fd8)] text-[var(--stat-pink,#ff4fd8)]";
  if (tone === "red") return "border-[var(--stat-wrong,#ff3030)] text-[var(--stat-wrong,#ff3030)]";
  if (tone === "white") return "border-[var(--nffc-white,#f5f5f5)] text-[var(--nffc-white,#f5f5f5)]";
  return "border-[var(--nffc-muted,#a7a7a7)] text-[var(--nffc-muted,#a7a7a7)]";
}

function getTextToneClass(tone: Tone | "red") {
  if (tone === "green") return "text-[var(--stat-green,#22e55e)]";
  if (tone === "yellow") return "text-[var(--stat-yellow,#ffe44d)]";
  if (tone === "cyan") return "text-[var(--stat-cyan,#59efff)]";
  if (tone === "pink") return "text-[var(--stat-pink,#ff4fd8)]";
  if (tone === "red") return "text-[var(--stat-wrong,#ff3030)]";
  if (tone === "white") return "text-[var(--nffc-white,#f5f5f5)]";
  return "text-[var(--nffc-muted,#a7a7a7)]";
}
